const handleSort = (field: "date" | "bp" | "tp" | "type") => {
  if (sortBy === field) {
    setSortDir((d) => (d === "asc" ? "desc" : "asc"));
  } else {
    setSortBy(field);
    setSortDir("desc");
  }
  setTxOffset(0);  // reset pagination on sort change
  setAllTx([]);    // clear accumulated rows
};

const sortIcon = (field: string) =>
