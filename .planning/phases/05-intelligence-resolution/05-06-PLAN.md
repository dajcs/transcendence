```typescript
const onBetResolved = (data: { bet_id: string; outcome: string; payout_summary: { winners: number; overturned: boolean } }) => {
  queryClient.invalidateQueries({ queryKey: ["market", marketId] });
  queryClient.invalidateQueries({ queryKey: ["resolution", marketId] });
  queryClient.invalidateQueries({ queryKey: ["positions"] });
  // Show payout banner (setPayoutBanner is already in component state from plan 05)
  const msg = data.payout_summary.overturned
    ? `Resolution overturned! Outcome: ${data.outcome.toUpperCase()}. ${data.payout_summary.winners} winner(s) paid.`
    : `Bet resolved: ${data.outcome.toUpperCase()}. ${data.payout_summary.winners} winner(s) paid.`;
  setPayoutBanner(msg);
  bootstrap();
};

const onDisputeOpened = () => {
  queryClient.invalidateQueries({ queryKey: ["resolution", marketId] });
  queryClient.invalidateQueries({ queryKey: ["market", marketId] });
};

const onDisputeVoted = (data: { bet_id: string; yes_weight: number; no_weight: number }) => {
  // Update resolution query cache directly for immediate tally update
  queryClient.setQueryData(
    ["resolution", marketId],
    (old: { resolution: unknown; dispute: { yes_weight: number; no_weight: number } | null } | undefined) =>
      old?.dispute
        ? { ...old, dispute: { ...old.dispute, yes_weight: data.yes_weight, no_weight: data.no_weight } }
        : old
  );
};

const onDisputeClosed = () => {
  queryClient.invalidateQueries({ queryKey: ["resolution", marketId] });
  queryClient.invalidateQueries({ queryKey: ["market", marketId] });
};
```

Register them (add after existing `socket.on("bet:comment_added", onCommentAdded)`):
```typescript
socket.on("bet:resolved", onBetResolved);
socket.on("dispute:opened", onDisputeOpened);
socket.on("dispute:voted", onDisputeVoted);
socket.on("dispute:closed", onDisputeClosed);
```

```typescript
socket.off("dispute:closed", onDisputeClosed);
```

IMPORTANT: All 4 new event listeners go INTO the existing single useEffect — do NOT create a second useEffect. The dependency array `[socket, marketId, queryClient]` already covers these.
